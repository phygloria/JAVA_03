/*
* 클래스 레벨에 @RequestMapping 어노테이션 사용이 가능하다.
* 클래스 레벨에 url은 공통부분을 이용해 설정하고 나면 매번 핸들러 메소드에 url의 중복되는 내용은 작성하지 않아도 된다.
* 이 때 와일드 카드를 이용해서 조금 더 포괄적인 url 패턴을 설정할 수 있다.
*/

@Controller //스프링빈의 이름 생략
@RequestMapping("/orders/")
public class ClassMappingTestController {

	@GetMapping("/") //가져오기 사용자가 입력한 url주소 그대로 처리
	public String registOrder(){
		return "hellow"; //서버바꾸면 항상실행
	}
	
	@GetMapping("/orders/test")
	public String regisTest(){
		return "test";
	}
	@GetMapping("/test")
	public String regisTest1(){
		return "test1";
	}

}
